  \section{Meta-Theory}
  \begin{frame}{Judgments}
   \begin{align*}
     \nonterm{judgment}
       & \ceq  \wflevel{\star} \\
       & \pipe \wfctx{\star} \\
       & \pipe \wfterm{\star} \\
       & \pipe \wfelim{\star} \\
       & \pipe \nonterm{ctx} \vdash \nonterm{judgment} \\
       & \pipe \BCheck{\nonterm{elim}}{\nonterm{term}} \\
       & \pipe \BSynth{\nonterm{term}}{\nonterm{term}} \\
   \end{align*}

   \begin{center}
     $\star$ = an arbitrary string of meta-variables and characters
   \end{center}

  \end{frame}
  \begin{frame}{Universes}
    \RightLabel{\Type formation}
    \AxiomC{\wflevel{\mv{n}}}
    \UnaryInfC{\wfterm{\type{\mv{n}}}}
    \DisplayProof
    \RightLabel{\Type introduction}
    \AxiomC{}
    \UnaryInfC{\BCheck{\type{\mrm{s}(\mv{n})}}{\type{\mv{n}}}}
    \DisplayProof
  \end{frame}
  \begin{frame}{True Type}
    \RightLabel{$\top$ formation}
    \AxiomC{}
    \UnaryInfC{$\mathbb{U}_0 \ni \top$}
    \DisplayProof
    \RightLabel{$\top$ introduction}
    \AxiomC{}
    \UnaryInfC{$\top \ni tt$}
    \DisplayProof
    \RightLabel{$\top$ elimination}
    \AxiomC{$\Gamma \vdash x : A$}
    \AxiomC{$\Gamma \vdash u : \top$}
    \BinaryInfC{$A \ni absorb_A(u,x)$}
    \DisplayProof
  \end{frame}
  \begin{frame}{False Type}
    \RightLabel{$\bot$ formation}
    \AxiomC{}
    \UnaryInfC{$\mathbb{U}_0 \ni \bot$}
    \DisplayProof
    \RightLabel{$\bot$ elimination}
    \AxiomC{$\Gamma \vdash A : \mathbb{U}_0$}
    \AxiomC{$\Gamma \vdash e : \bot$}
    \BinaryInfC{$A \ni absurd_A(e)$}
    \DisplayProof
  \end{frame}
  \begin{frame}{Conjunction Type}

  \end{frame}
